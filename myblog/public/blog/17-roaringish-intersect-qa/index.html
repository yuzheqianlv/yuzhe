fn naive_intersect(a: &[u32], b: &[u32]) -> Vec<u32> {
    let mut result = Vec::new();
    let mut i = 0;
    let mut j = 0;
    
    while i < a.len() && j < b.len() {
        match a[i].cmp(&b[j]) {
            Equal => {
                result.push(a[i]);
                i += 1;
                j += 1;
            }
            Less => i += 1,
            Greater => j += 1,
        }
    }
    result
}
